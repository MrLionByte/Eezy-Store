User {
        id int PK
        email string
        password string
        first_name string
        last_name string
        is_staff boolean
        is_active boolean
        date_joined datetime
    }
    
Product {
        id int PK
        name string
        description text
        price decimal
        image string
        created_at datetime
        updated_at datetime
    }
    
Rating {
        id int PK
        product_id int FK
        user_id int FK
        score int
        created_at datetime
    }
    
Address {
        id int PK
        user_id int FK
        street string
        city string
        state string
        postal_code string
        country string
        is_default boolean
    }
    
Order {
        id int PK
        user_id int FK
        address_id int FK
        status string
        total_amount decimal
        created_at datetime
        updated_at datetime
    }
    
OrderItem {
        id int PK
        order_id int FK
        product_id int FK
        quantity int
        price decimal
    }
    
Cart {
        id int PK
        user_id int FK
        created_at datetime
        updated_at datetime
    }
    
CartItem {
        id int PK
        cart_id int FK
        product_id int FK
        quantity int
    }
    
User ||--o{ Address : has
User ||--o{ Order : places
User ||--o{ Rating : gives
User ||--o{ Cart : has
    
Product ||--o{ Rating : receives
Product ||--o{ OrderItem : "ordered in"
Product ||--o{ CartItem : "added to"
    
Order ||--o{ OrderItem : contains
Order ||--o{ Address : "ships to"
    
Cart ||--o{ CartItem : contains
